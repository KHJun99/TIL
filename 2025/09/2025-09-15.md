def dfs(v, visited, graph):
    visited[v] = True
    print(v, end=" ")
    for nxt in graph[v]:
        if not visited[nxt]:
            dfs(nxt, visited, graph)

# BFS (í)
from collections import deque
def bfs(start, graph):
    visited = [False] * len(graph)
    q = deque([start])
    visited[start] = True
    while q:
        v = q.popleft()
        print(v, end=" ")
        for nxt in graph[v]:
            if not visited[nxt]:
                visited[nxt] = True
                q.append(nxt)

# Union-Find

def union(x, y):
